using System; using System.Collections.Generic; namespace CricketCareerGame { class Program { static void Main(string[] args) { GameManager game = new GameManager(); game.StartCareer(); } } // ---------------- PLAYER ---------------- public enum Role { Batsman, Bowler, AllRounder, Keeper } public class Player { public string Name; public Role Role; public int Batting; public int Bowling; public int Fitness; public float Form = 50; public bool Injured = false; public Player(string name, Role role) { Name = name; Role = role; Batting = 50; Bowling = 50; Fitness = 50; } } // ---------------- MATCH ENGINE ---------------- public class BallResult { public int Runs; public bool Wicket; public string Type; } public class MatchEngine { Random rand = new Random(); public BallResult PlayBall(Player batsman, Player bowler) { BallResult r = new BallResult(); double skill = batsman.Batting + batsman.Form - bowler.Bowling; double outcome = rand.NextDouble() * 100 + skill; if (outcome > 120) r.Runs = 6; else if (outcome > 100) r.Runs = 4; else if (outcome > 70) r.Runs = rand.Next(1, 3); else if (outcome < 30) { r.Wicket = true; string[] types = { "Bowled", "Caught", "LBW", "RunOut" }; r.Type = types[rand.Next(types.Length)]; } return r; } } // ---------------- MATCH ---------------- public class Match { public int Runs = 0; public int Wickets = 0; public int Balls = 0; public void Play(Player batsman, Player bowler) { MatchEngine engine = new MatchEngine(); while (Balls < 120 && Wickets < 10) { BallResult r = engine.PlayBall(batsman, bowler); Balls++; if (r.Wicket) { Wickets++; Console.WriteLine($"WICKET! {r.Type}"); } else { Runs += r.Runs; Console.WriteLine($"Runs: {r.Runs}"); } } Console.WriteLine($"Final Score: {Runs}/{Wickets}"); } } // ---------------- CAREER ---------------- public class Career { public Player Player; public int XP = 0; public int Level = 1; public Career(Player p) { Player = p; } public void AddXP(int amount) { XP += amount; if (XP > Level * 100) { Level++; XP = 0; Upgrade(); } } void Upgrade() { Player.Batting += 5; Player.Bowling += 5; Player.Fitness += 2; Console.WriteLine("Level Up!"); } } // ---------------- AI ---------------- public class AI { public Player CreateOpponent() { return new Player("AI Bowler", Role.Bowler); } } // ---------------- AUCTION ---------------- public class Auction { public int GetValue(Player p) { return p.Batting * 1000 + (int)p.Form * 500; } } // ---------------- INJURY ---------------- public class InjurySystem { Random rand = new Random(); public void Check(Player p) { if (rand.NextDouble() < 0.1) { p.Injured = true; Console.WriteLine($"{p.Name} got injured!"); } } } // ---------------- GAME MANAGER ---------------- public class GameManager { public void StartCareer() { Player user = new Player("You", Role.Batsman); Career career = new Career(user); AI ai = new AI(); Console.WriteLine("Career Started!"); for (int i = 0; i < 5; i++) { Console.WriteLine($"\nMatch {i + 1}"); Player opponent = ai.CreateOpponent(); Match match = new Match(); match.Play(user, opponent); int xpEarned = match.Runs; career.AddXP(xpEarned); new InjurySystem().Check(user); Console.WriteLine($"XP: {career.XP} | Level: {career.Level}"); } Auction auction = new Auction(); Console.WriteLine($"Player Market Value: {auction.GetValue(user)}"); } } }
